#SIMPLESCRIPT     

     SET $myid  $argv1 
     SET $nxtid $argv2
     SET $prvid $argv3
     SET $cnt   $argv4
     SET $time  $argv5
     
     BGT $myid 0 #L1
     PRN $myid "sending value 1 to " $nxtid
     SND $nxtid 1
     SUB $cnt $cnt 1
     
#L1  RCV $prvid $val
     PRN $myid "received value " $val " from " $prvid
     
     BEQ $cnt 0 #L2
   
     SLP $time
   
     ADD $val $val 1
     PRN $myid "sending value " $val " to " $nxtid
     SND $nxtid $val
     
     SUB $cnt $cnt 1
     BEQ $cnt 0 #L2
     
     BRA #L1

#L2  RET
